  <name>Task 3: Routing — / redirects unauthenticated users to /landing</name>
  <files>app/src/app/router/RootRedirect.tsx, app/src/app/router/AppRouter.tsx</files>
  <action>
    **Step 1 — Create app/src/app/router/RootRedirect.tsx (new file):**
    ```tsx
    import { Navigate } from 'react-router-dom';
    import { useAuth } from '../providers/AuthProvider';
    import { ROUTES } from './routeConfig';

    export function RootRedirect() {
      const { user, loading } = useAuth();
      if (loading) return null;
      // Unauthenticated → marketing landing page
      // Authenticated → main app (reports as first real page)
      return <Navigate to={user ? ROUTES.REPORTS : ROUTES.LANDING} replace />;
    }
    ```

    **Step 2 — app/src/app/router/AppRouter.tsx:**
    Read the file. It uses createBrowserRouter with an array. Add the "/" route as the FIRST entry (before /landing and *):
    ```tsx
    import React from 'react';
    import { createBrowserRouter, RouterProvider } from 'react-router-dom';
    import App from '../../App';
    import { OrganizationProvider } from '../providers';
    import LandingPage from '../../features/landing/LandingPage';
    import GuestRoute from './GuestRoute';
    import { RootRedirect } from './RootRedirect';

    const router = createBrowserRouter([
      {
        path: '/',
        element: <RootRedirect />,
      },
      {
        path: '/landing',
        element: <GuestRoute><LandingPage /></GuestRoute>,
      },
      {
        path: '*',
        element: <OrganizationProvider><App /></OrganizationProvider>,
      },
    ]);

    export default function AppRouter() {
      return <RouterProvider router={router} />;
    }
    ```

    Why this works:
    - Unauthenticated user at /: RootRedirect → Navigate to /landing ✓
    - Authenticated user at /: RootRedirect → Navigate to /reports ✓
    - GuestRoute at /landing: if auth user tries /landing directly → redirects to ROUTES.DASHBOARD (/) → RootRedirect → /reports ✓ (no infinite loop because / now goes to /reports for auth users)
    - All other app paths (*, /projects, /reports etc): unchanged — go to App ✓

    Avoid: changing GuestRoute.tsx, ProtectedRoute.tsx, or routeConfig.ts.
