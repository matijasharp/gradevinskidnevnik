  <action>

    Run: `cd app && npm install gsap`
    Confirm gsap appears in app/package.json under "dependencies".

    **Step 2 — Create app/src/features/auth/components/SplashScreen.tsx:**
    ```tsx
    import { useEffect, useRef } from 'react';
    import gsap from 'gsap';

    interface SplashScreenProps {
      onComplete: () => void;
    }

    export function SplashScreen({ onComplete }: SplashScreenProps) {
      const containerRef = useRef<HTMLDivElement>(null);
      const logoRef = useRef<HTMLImageElement>(null);
      const textRef = useRef<HTMLParagraphElement>(null);

      useEffect(() => {
        const tl = gsap.timeline({
          onComplete: () => {
            gsap.to(containerRef.current, {
              opacity: 0,
              duration: 0.4,
              delay: 0.5,
              onComplete,
            });
          },
        });

        tl.fromTo(
          logoRef.current,
          { opacity: 0, y: 24 },
          { opacity: 1, y: 0, duration: 0.6, ease: 'power2.out' }
        ).fromTo(
          textRef.current,
          { opacity: 0, y: 12 },
          { opacity: 1, y: 0, duration: 0.5, ease: 'power2.out' },
          '-=0.3'
        );

        return () => { tl.kill(); };
      }, [onComplete]);

      return (
        <div
          ref={containerRef}
          className="fixed inset-0 z-50 flex flex-col items-center justify-center bg-white"
        >
          <img
            ref={logoRef}
            src="/brand/logo.svg"
            alt="Građevinski Dnevnik Online"
            className="h-20 w-20 mb-5"
            style={{ opacity: 0 }}
          />
          <p
            ref={textRef}
            className="text-xl font-semibold text-gray-800 tracking-tight"
            style={{ opacity: 0, fontFamily: 'Jost, sans-serif' }}
          >
            Građevinski Dnevnik Online
          </p>
        </div>
      );
    }
    ```

    Avoid: importing useAuth, useNavigate, or any routing/auth logic into SplashScreen.
    Avoid: reading from DOM beyond the refs — GSAP targets only the refs.
  </action>

    Make these changes:

    1. Add import at top:
       ```tsx
       import { useState } from 'react'; // if not already imported
       import { SplashScreen } from './SplashScreen';
       ```

    2. Add state inside the component function body (after useAuth destructuring):
       ```tsx
       const [splashDone, setSplashDone] = useState(false);
       ```

    3. Find the loading state return (the `if (loading) return null;` or equivalent).
       Replace it with:
       ```tsx
       if (loading || !splashDone) {
         return <SplashScreen onComplete={() => setSplashDone(true)} />;
       }
       ```

    This creates the sequence:
    - Mount → loading=true, splashDone=false → SplashScreen renders and animates
    - GSAP timeline ends → onComplete → setSplashDone(true)
    - If loading still true: SplashScreen renders (because !splashDone is now false but loading is still true — actually wait: `loading || !splashDone` — if splashDone=true and loading=false, condition is false → renders login/app ✓. If splashDone=true and loading=true, condition is true → still shows splash (blank, opacity 0 since GSAP already faded it out) — acceptable brief flash during slow auth.

    IMPORTANT: Use `useState` import — check if it's already imported (React components often import it). Only add if missing.

    Avoid: changing any auth callback functions, form submit handlers, or other state variables.
    Avoid: removing the existing loading check — only REPLACE it with the SplashScreen render.
