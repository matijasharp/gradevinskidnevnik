    ```ts
    export const searchOrganizations = async (query: string): Promise<Company[]> => {
      if (query.trim().length < 2) return [];
      ensureSupabase();
      const { data } = await supabase
        .from('organizations')
        .select('*')
        .ilike('name', `%${query.trim()}%`)
        .limit(10);
      return (data ?? []).map(mapOrganization);
    };
    ```

    - Returns up to 10 matching orgs
    - Guard: returns [] if query < 2 chars (no round-trip for empty/single char)
    - No new interfaces needed — returns existing `Company[]`
